CREATE DATABASE IF NOT EXISTS learning_portal
  CHARACTER SET utf8mb4
  COLLATE utf8mb4_unicode_ci;

USE learning_portal;


-- ─── USERS ───────────────────────────────────────────────────────────────────
CREATE TABLE users (
    user_id      INT AUTO_INCREMENT PRIMARY KEY,
    username     VARCHAR(50)  UNIQUE NOT NULL,
    email        VARCHAR(120) UNIQUE NOT NULL,
    password_hash VARCHAR(256) NOT NULL,
    role         ENUM('student','instructor','admin') DEFAULT 'student',
    full_name    VARCHAR(120),
    profile_pic  TEXT,
    bio          TEXT,
    is_active    TINYINT(1) DEFAULT 1,
    created_at   DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at   DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

-- ─── CATEGORIES ──────────────────────────────────────────────────────────────
CREATE TABLE categories (
    category_id  INT AUTO_INCREMENT PRIMARY KEY,
    name         VARCHAR(80) UNIQUE NOT NULL,
    slug         VARCHAR(80) UNIQUE NOT NULL,
    description  TEXT,
    icon         VARCHAR(50)
);







-- ─── COURSES ─────────────────────────────────────────────────────────────────
CREATE TABLE courses (
    course_id     INT AUTO_INCREMENT PRIMARY KEY,
    instructor_id INT NOT NULL,
    category_id   INT NOT NULL,
    title         VARCHAR(200) NOT NULL,
    description   TEXT,
    price         DECIMAL(8,2) DEFAULT 0.00,
    thumbnail     TEXT,
    level         ENUM('beginner','intermediate','advanced') DEFAULT 'beginner',
    status        ENUM('draft','published','archived') DEFAULT 'draft',
    language      VARCHAR(30) DEFAULT 'English',
    rating        FLOAT DEFAULT 0.0,
    total_students INT DEFAULT 0,
    created_at    DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at    DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (instructor_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (category_id)   REFERENCES categories(category_id)
);

-- ─── LESSONS ─────────────────────────────────────────────────────────────────
CREATE TABLE lessons (
    lesson_id    INT AUTO_INCREMENT PRIMARY KEY,
    course_id    INT NOT NULL,
    title        VARCHAR(200) NOT NULL,
    content      LONGTEXT,
    video_url    TEXT,
    duration_min INT DEFAULT 0,
    order_no     INT NOT NULL DEFAULT 1,
    is_free      TINYINT(1) DEFAULT 0,
    created_at   DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (course_id) REFERENCES courses(course_id) ON DELETE CASCADE
);

-- ─── ENROLLMENTS ─────────────────────────────────────────────────────────────
CREATE TABLE enrollments (
    enrollment_id  INT AUTO_INCREMENT PRIMARY KEY,
    user_id        INT NOT NULL,
    course_id      INT NOT NULL,
    amount_paid    DECIMAL(8,2) DEFAULT 0.00,
    payment_status ENUM('free','paid','refunded') DEFAULT 'free',
    payment_id     VARCHAR(100),
    enrolled_at    DATETIME DEFAULT CURRENT_TIMESTAMP,
    completed_at   DATETIME,
    UNIQUE KEY uq_enroll (user_id, course_id),
    FOREIGN KEY (user_id)   REFERENCES users(user_id)   ON DELETE CASCADE,
    FOREIGN KEY (course_id) REFERENCES courses(course_id) ON DELETE CASCADE
);

-- ─── PROGRESS ────────────────────────────────────────────────────────────────
CREATE TABLE progress (
    progress_id   INT AUTO_INCREMENT PRIMARY KEY,
    enrollment_id INT NOT NULL,
    lesson_id     INT NOT NULL,
    is_completed  TINYINT(1) DEFAULT 0,
    completed_at  DATETIME,
    UNIQUE KEY uq_progress (enrollment_id, lesson_id),
    FOREIGN KEY (enrollment_id) REFERENCES enrollments(enrollment_id) ON DELETE CASCADE,
    FOREIGN KEY (lesson_id)     REFERENCES lessons(lesson_id)         ON DELETE CASCADE
);

-- ─── QUIZZES ─────────────────────────────────────────────────────────────────
CREATE TABLE quizzes (
    quiz_id     INT AUTO_INCREMENT PRIMARY KEY,
    course_id   INT NOT NULL,
    title       VARCHAR(200) NOT NULL,
    description TEXT,
    pass_score  INT DEFAULT 70,
    time_limit  INT DEFAULT 0,
    FOREIGN KEY (course_id) REFERENCES courses(course_id) ON DELETE CASCADE
);

CREATE TABLE quiz_questions (
    question_id   INT AUTO_INCREMENT PRIMARY KEY,
    quiz_id       INT NOT NULL,
    question_text TEXT NOT NULL,
    question_type ENUM('mcq','true_false','short') DEFAULT 'mcq',
    marks         INT DEFAULT 1,
    order_no      INT DEFAULT 1,
    FOREIGN KEY (quiz_id) REFERENCES quizzes(quiz_id) ON DELETE CASCADE
);

CREATE TABLE quiz_options (
    option_id    INT AUTO_INCREMENT PRIMARY KEY,
    question_id  INT NOT NULL,
    option_text  VARCHAR(500) NOT NULL,
    is_correct   TINYINT(1) DEFAULT 0,
    FOREIGN KEY (question_id) REFERENCES quiz_questions(question_id) ON DELETE CASCADE
);

CREATE TABLE quiz_attempts (
    attempt_id    INT AUTO_INCREMENT PRIMARY KEY,
    quiz_id       INT NOT NULL,
    user_id       INT NOT NULL,
    score         INT DEFAULT 0,
    total_marks   INT DEFAULT 0,
    passed        TINYINT(1) DEFAULT 0,
    attempted_at  DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (quiz_id)  REFERENCES quizzes(quiz_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id)  REFERENCES users(user_id)   ON DELETE CASCADE
);

-- ─── REVIEWS ─────────────────────────────────────────────────────────────────
CREATE TABLE reviews (
    review_id  INT AUTO_INCREMENT PRIMARY KEY,
    user_id    INT NOT NULL,
    course_id  INT NOT NULL,
    rating     INT CHECK (rating BETWEEN 1 AND 5),
    comment    TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    UNIQUE KEY uq_review (user_id, course_id),
    FOREIGN KEY (user_id)   REFERENCES users(user_id)     ON DELETE CASCADE,
    FOREIGN KEY (course_id) REFERENCES courses(course_id) ON DELETE CASCADE
);

-- ─── CERTIFICATES ────────────────────────────────────────────────────────────
CREATE TABLE certificates (
    cert_id     INT AUTO_INCREMENT PRIMARY KEY,
    user_id     INT NOT NULL,
    course_id   INT NOT NULL,
    cert_number VARCHAR(50) UNIQUE NOT NULL,
    issued_at   DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id)   REFERENCES users(user_id)     ON DELETE CASCADE,
    FOREIGN KEY (course_id) REFERENCES courses(course_id) ON DELETE CASCADE
);

CREATE OR REPLACE PROCEDURE SP_ERROR (P_PROGRAM VARCHAR2,
P_ERROR_NO VARCHAR2,P_ERROR_MSG VARCHAR2)
IS
PRAGMA AUTONOMOUS_TRANSACTION;

INSERT INTO ERROR_LOG VALUES (P_MODULE, SYSDATE,P_ERROR_NO, P_ERROR_MSG);

END;


CREATE OR REPLACE PROCEDURE pr_register_user (
    p_full_name   IN users.full_name%TYPE,
    p_email       IN users.email%TYPE,
    p_phone       IN users.phone%TYPE,
    p_user_type   IN users.user_type%TYPE
)
IS
    v_count NUMBER;
BEGIN
    SELECT COUNT(*)
    INTO v_count
    FROM users
    WHERE email = p_email AND phone= p_phone;

    IF v_count > 0 THEN
        RAISE_APPLICATION_ERROR(-20001, 'Email/Phone_no already exists');
    END IF;

    INSERT INTO users (user_id, full_name, email, user_type, created_at)
    VALUES (seq_users.NEXTVAL, p_full_name, p_email, UPPER(p_user_type), SYSDATE);
    COMMIT;

    DBMS_OUTPUT.PUT_LINE('User registered successfully');
EXCEPTION
WHEN OTHERS THEN
SP_ERROR_LOG ('pr_register_user',SQLCODE,SQLERRM,
SUBSTR (DBMS_UTILITY.format_error_backtrace,-5));
RAISE_APPLICATION_ERROR(-20400,SQLERRM);
END;
END;

CREATE OR REPLACE PROCEDURE pr_create_course (
    p_course_name    IN courses.course_name%TYPE,
    p_instructor_id  IN courses.instructor_id%TYPE,
    p_price          IN courses.price%TYPE,
    p_max_students   IN courses.max_students%TYPE
)
IS
    v_count NUMBER;
BEGIN
    SELECT COUNT(*)
    INTO v_count
    FROM users
    WHERE user_id = p_instructor_id
      AND user_type = 'INSTRUCTOR';

    IF v_count = 0 THEN
        RAISE_APPLICATION_ERROR(-20002, 'Invalid instructor');
    END IF;

    INSERT INTO courses (
        course_id, course_name, instructor_id, price,
        max_students, enrolled_count, course_status, created_at
    )
    VALUES (
        seq_courses.NEXTVAL, p_course_name, p_instructor_id, p_price,
        p_max_students, 0, 'ACTIVE', SYSDATE
    );
COMMIT;

    DBMS_OUTPUT.PUT_LINE('Course created successfully');
EXCEPTION
WHEN OTHERS THEN
SP_ERROR_LOG ('pr_create_course',SQLCODE,SQLERRM,
SUBSTR (DBMS_UTILITY.format_error_backtrace,-5));
RAISE_APPLICATION_ERROR(-20400,SQLERRM);
END;
END;

CREATE OR REPLACE PROCEDURE pr_enroll_student (
    p_user_id    IN enrollments.user_id%TYPE,
    p_course_id  IN enrollments.course_id%TYPE
)
IS
    v_user_count       NUMBER;
    v_course_count     NUMBER;
    v_dup_count        NUMBER;
    v_max_students     courses.max_students%TYPE;
    v_enrolled_count   courses.enrolled_count%TYPE;
BEGIN
    SELECT COUNT(*)
    INTO v_user_count
    FROM users
    WHERE user_id = p_user_id
      AND user_type = 'STUDENT';

    IF v_user_count = 0 THEN
        RAISE_APPLICATION_ERROR(-20003, 'Student not found');
    END IF;

    SELECT COUNT(*)
    INTO v_course_count
    FROM courses
    WHERE course_id = p_course_id
      AND course_status = 'ACTIVE';

    IF v_course_count = 0 THEN
        RAISE_APPLICATION_ERROR(-20004, 'Course not available');
    END IF;

    SELECT COUNT(*)
    INTO v_dup_count
    FROM enrollments
    WHERE user_id = p_user_id
      AND course_id = p_course_id;

    IF v_dup_count > 0 THEN
        RAISE_APPLICATION_ERROR(-20005, 'Student already enrolled');
    END IF;

    SELECT max_students, enrolled_count
    INTO v_max_students, v_enrolled_count
    FROM courses
    WHERE course_id = p_course_id;

    IF v_enrolled_count >= v_max_students THEN
        RAISE_APPLICATION_ERROR(-20006, 'Course is full');
    END IF;

    INSERT INTO enrollments (
        enrollment_id, user_id, course_id, enrollment_date,
        completion_status, progress_percent, payment_status
    )
    VALUES (
        seq_enrollments.NEXTVAL, p_user_id, p_course_id, SYSDATE,
        'IN_PROGRESS', 0, 'PENDING'
    );
COMMIT;

    DBMS_OUTPUT.PUT_LINE('Enrollment successful');

EXCEPTION
WHEN OTHERS THEN
SP_ERROR_LOG ('pr_enroll_student',SQLCODE,SQLERRM,
SUBSTR (DBMS_UTILITY.format_error_backtrace,-5));
RAISE_APPLICATION_ERROR(-20400,SQLERRM);
END;

END;

---- procedure for  progress of the  register


CREATE OR REPLACE PROCEDURE pr_update_progress (
    p_user_id           IN enrollments.user_id%TYPE,
    p_course_id         IN enrollments.course_id%TYPE,
    p_progress_percent  IN NUMBER
)
IS
BEGIN
    UPDATE enrollments
    SET progress_percent = p_progress_percent,
        completion_status = CASE
                                WHEN p_progress_percent >= 100 THEN 'COMPLETED'
                                ELSE 'IN_PROGRESS'
                            END
    WHERE user_id = p_user_id
      AND course_id = p_course_id;

    DBMS_OUTPUT.PUT_LINE('Progress updated');
EXCEPTION
WHEN OTHERS THEN
SP_ERROR_LOG ('pr_update_progress',SQLCODE,SQLERRM,
SUBSTR (DBMS_UTILITY.format_error_backtrace,-5));
RAISE_APPLICATION_ERROR(-20400,SQLERRM);
END;

END;

---function for certificate  eligibility

CREATE OR REPLACE FUNCTION fn_certificate_eligible (
    p_user_id   IN NUMBER,
    p_course_id IN NUMBER
)
RETURN VARCHAR2
IS
    v_completion enrollments.completion_status%TYPE;
    v_pass_count NUMBER;
BEGIN
    SELECT completion_status
    INTO v_completion
    FROM enrollments
    WHERE user_id = p_user_id
      AND course_id = p_course_id;

    SELECT COUNT(*)
    INTO v_pass_count
    FROM results
    WHERE user_id = p_user_id
      AND course_id = p_course_id
      AND result_status = 'PASS';

    IF v_completion = 'COMPLETED' AND v_pass_count > 0 THEN
        RETURN 'ELIGIBLE';
    ELSE
        RETURN 'NOT ELIGIBLE';
    END IF;
EXCEPTION
    WHEN NO_DATA_FOUND THEN
        RETURN 'NOT ELIGIBLE';
WHEN OTHERS THEN
SP_ERROR_LOG ('SP_COMM_CAL',SQLCODE,SQLERRM,
SUBSTR (DBMS_UTILITY.format_error_backtrace,-5));
RAISE_APPLICATION_ERROR(-20400,SQLERRM);
END;

END;

        p_user_id     IN NUMBER,

CREATE OR REPLACE PACKAGE BODY pkg_lms AS

    PROCEDURE register_user (
        p_full_name   IN VARCHAR2,
        p_email       IN VARCHAR2,
        p_user_type   IN VARCHAR2
    )
    IS
        v_count NUMBER;
    BEGIN
        SELECT COUNT(*)
        INTO v_count
        FROM users
        WHERE email = p_email;

        IF v_count > 0 THEN
            RAISE_APPLICATION_ERROR(-20101, 'Email already exists');
        END IF;

        INSERT INTO users (user_id, full_name, email, user_type, created_at)
        VALUES (seq_users.NEXTVAL, p_full_name, p_email, UPPER(p_user_type), SYSDATE);
    END register_user;

    PROCEDURE enroll_student (
        p_user_id     IN NUMBER,
        p_course_id   IN NUMBER
    )
    IS
        v_dup_count NUMBER;
    BEGIN
        SELECT COUNT(*)
        INTO v_dup_count
        FROM enrollments
        WHERE user_id = p_user_id
          AND course_id = p_course_id;

        IF v_dup_count > 0 THEN
            RAISE_APPLICATION_ERROR(-20102, 'Already enrolled');
        END IF;

        INSERT INTO enrollments (
            enrollment_id, user_id, course_id, enrollment_date,
            completion_status, progress_percent, payment_status
        )
        VALUES (
            seq_enrollments.NEXTVAL, p_user_id, p_course_id, SYSDATE,
            'IN_PROGRESS', 0, 'PENDING'
        );
    END enroll_student;

    FUNCTION get_course_fee (
        p_course_id IN NUMBER
    )
    RETURN NUMBER
    IS
        v_fee NUMBER;
    BEGIN
        SELECT price INTO v_fee
        FROM courses
        WHERE course_id = p_course_id;

        RETURN v_fee;
    EXCEPTION
        WHEN NO_DATA_FOUND THEN
            RETURN 0;
    END get_course_fee;

    FUNCTION certificate_eligible (
        p_user_id   IN NUMBER,
        p_course_id IN NUMBER
    )
    RETURN VARCHAR2
    IS
        v_status VARCHAR2(20);
    BEGIN
        v_status := fn_certificate_eligible(p_user_id, p_course_id);
        RETURN v_status;
    END certificate_eligible;

END pkg_lms;

---- trigger for  duplicate enroll
CREATE OR REPLACE TRIGGER trg_prevent_duplicate_enroll
BEFORE INSERT ON enrollments
FOR EACH ROW
DECLARE
    v_count NUMBER;
BEGIN
    SELECT COUNT(*)
    INTO v_count
    FROM enrollments
    WHERE user_id = :NEW.user_id
      AND course_id = :NEW.course_id;

    IF v_count > 0 THEN
        RAISE_APPLICATION_ERROR(-20201, 'Duplicate enrollment not allowed');
    END IF;
EXCEPTION
WHEN OTHERS THEN
SP_ERROR_LOG ('SP_COMM_CAL',SQLCODE,SQLERRM,
SUBSTR (DBMS_UTILITY.format_error_backtrace,-5));
RAISE_APPLICATION_ERROR(-20400,SQLERRM);
END;

END;

    WHERE course_id = :NEW.course_id;
EXCEPTION
WHEN OTHERS THEN
SP_ERROR_LOG ('SP_COMM_CAL',SQLCODE,SQLERRM,
SUBSTR (DBMS_UTILITY.format_error_backtrace,-5));
RAISE_APPLICATION_ERROR(-20400,SQLERRM);
END;

END;

-----trigger for result
CREATE OR REPLACE TRIGGER trg_result_status
BEFORE INSERT OR UPDATE ON results
FOR EACH ROW
DECLARE
    v_pass_marks quiz.pass_marks%TYPE;
BEGIN
    SELECT pass_marks
    INTO v_pass_marks
    FROM quiz
    WHERE quiz_id = :NEW.quiz_id;

    IF :NEW.score >= v_pass_marks THEN
        :NEW.result_status := 'PASS';
    ELSE
        :NEW.result_status := 'FAIL';
    END IF;
EXCEPTION
WHEN OTHERS THEN
SP_ERROR_LOG ('SP_COMM_CAL',SQLCODE,SQLERRM,
SUBSTR (DBMS_UTILITY.format_error_backtrace,-5));
RAISE_APPLICATION_ERROR(-20400,SQLERRM);
END;

END;
